uint8_t cal_table_high_first(uint8_t value)
{
	uint8_t i, crc;

	crc = value;
	/* 数据往左移了8位，需要计算8次 */
	for (i = 8; i>0; --i)
	{
		if (crc & 0x80)  /* 判断最高位是否为1 */
		{
			/* 最高位为1，不需要异或，往左移一位，然后与0x07异或 */
			/* 0x07(多项式：x8+x2+x+1， 100001101)，最高位不需要异或，直接去掉 */
			crc = (crc << 1) ^ 0x07;
		}
		else
		{
			/* 最高位为0时，不需要异或，整体数据往左移一位 */
			crc = (crc << 1);
		}
	}

	return crc;
}

uint8_t  Calculate_CRC8(uint8_t u8Array[], uint8_t u8ArrayLength)
{
	uint8_t  i, j;
	uint8_t  u8crc = 0x00;
	uint8_t  u8CRC_Key = 0x07;//Polynomial

	/* 0x07 (多项式：x8+x2+x1+1，1 0000 0111)，最高位不需要异或，直接去掉 */
	for (j = 0; j < u8ArrayLength; j++)
	{
		/* 每次先与需要计算的数据异或,计算完指向下一数据 */
		u8crc ^= u8Array[j];
		/* 下面这段计算过程与计算一个字节crc一样 */
		for (i = 0; i < 8; i++)
		{
			if ((u8crc & 0x80) == 0x80)
			{
				u8crc = (uint8_t)(((uint8_t)u8crc << 1) ^ (uint8_t)u8CRC_Key);
			}
			else
			{
				u8crc = (uint8_t)(u8crc << 1);
			}
		}
	}
	return u8crc;
}

uint8_t CRC8_32_Table(uint8_t u8Data, uint8_t u8LastPEC)
{

	const uint8_t u8CRC_HorizontalArray[] = {
		0x00, 0x70, 0xE0, 0x90, 0xC7, 0xB7, 0x27, 0x57, 0x89, 0xF9, 0x69, 0x19, 0x4E, 0x3E, 0xAE, 0xDE
	};
	const uint8_t u8CRC_VerticalArray[] = {
		0x00, 0x07, 0x0E, 0x09, 0x1C, 0x1B, 0x12, 0x15, 0x38, 0x3F, 0x36, 0x31, 0x24, 0x23, 0x2A, 0x2D
	};

	uint8_t u8HighVal, u8LowVal;
	uint8_t u8Value;

	u8LastPEC ^= u8Data;
	u8HighVal = u8LastPEC >> 4;
	u8LowVal = u8LastPEC & 0x0F;
	u8Value = u8CRC_HorizontalArray[u8HighVal];
	u8Value ^= u8CRC_VerticalArray[u8LowVal];

	return u8Value;
}
